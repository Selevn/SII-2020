bind uint func FindFactorial(uint a){
extern uint func ord(char i);
extern char func chr(uint i);
	bind uint answer;
	answer = 1;
	while(a>1)[answer = answer*a; a = a-1;];
	stop answer;
};

output "Нахождение факториала числа ";
extern uint func random(uint a);
bind uint number;
number = random(11);
output number;
output "Равен ";
number = FindFactorial(number);
output number;
